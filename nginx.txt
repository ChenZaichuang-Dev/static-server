worker_processes 1;
worker_cpu_affinity 1;
worker_rlimit_nofile 8;

events {
    worker_connections 2;
    use epoll;
}

http {
    include mime.types;

    default_type application/octet-stream;

    access_log off;
    error_log /dev/null crit;

    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;

    keepalive_timeout 65;
    keepalive_requests 100;

    gzip on;
    gzip_disable "msie6";

    server {
        listen 80;
        server_name localhost;

        root /var/www/html;

        index index.html;

        location / {
            try_files $uri $uri/ =404;
        }
    }
}
